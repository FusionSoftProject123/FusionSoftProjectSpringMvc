  CREATE TABLE "FUSIONSOFTUSERS" 
   (	"USERID" NUMBER NOT NULL ENABLE, 
	"USERNAME" VARCHAR2(50 BYTE) NOT NULL ENABLE, 
	"FIRSTNAME" VARCHAR2(50 BYTE), 
	"LASTNAME" VARCHAR2(50 BYTE), 
	"PROFESSION" VARCHAR2(20 BYTE), 
	"PASSWORD" VARCHAR2(20 BYTE)
   ) ;

  CREATE OR REPLACE TRIGGER "FUSIONSOFTUSERS_TRG" 
BEFORE INSERT ON FUSIONSOFTUSERS 
FOR EACH ROW 
BEGIN
  <<COLUMN_SEQUENCES>>
  BEGIN
    IF INSERTING THEN
      SELECT FUSIONSOFTUSERS_SEQ.NEXTVAL INTO :NEW.USERID FROM SYS.DUAL;
    END IF;
  END COLUMN_SEQUENCES;
END;

ALTER TRIGGER "FUSIONSOFTUSERS_TRG" ENABLE;
